import sqlite3


conn = sqlite3.connect('standent13.db')
cursor = conn.cursor()

cursor.execute('''
CREATE TABLE IF NOT EXISTS "Group" (
    id INTEGER PRIMARY KEY,
    name TEXT 
)
''')

cursor.execute('''
CREATE TABLE IF NOT EXISTS Student (
    id INTEGER PRIMARY KEY,
    f TEXT,
    i TEXT,
    o TEXT,
    groupid INTEGER,
    FOREIGN KEY (groupid) REFERENCES "Group"(id)
)
''')
